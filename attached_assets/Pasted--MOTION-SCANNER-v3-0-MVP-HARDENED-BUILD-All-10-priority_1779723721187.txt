## 1. `qualification.py` — COMPLETE (TriState + all v2.0 gates + new options/sector gates)

```python
"""
Tri-state qualification engine. Returns GO / HOLD / ABORT with explicit reason.

Gate order (fail-fast):
  1. Data integrity      → ABORT
  2. Earnings blackout   → ABORT
  3. Liquidity floor     → ABORT
  4. Price/RVOL/ATR core → HOLD
  5. EMA stack / RSI band→ HOLD
  6. Indicator gates     → HOLD
  7. Options flow gate   → HOLD
  8. Sector headwind veto→ HOLD
  9. All pass            → GO
"""
from __future__ import annotations
from dataclasses import dataclass
from typing import Optional, Dict


@dataclass
class TriState:
    state: str   # "GO" | "HOLD" | "ABORT"
    reason: str

    def __iter__(self):
        return iter((self.state, self.reason))


def qualify(tech: Dict, fund: Dict, flow: Dict, cfg: Dict,
            indicators: Optional[Dict] = None,
            options: Optional[Dict] = None,
            sector_ctx: Optional[Dict] = None) -> TriState:

    # ── 1. Data integrity ────────────────────────────────────────────────
    if not tech or not tech.get("ok"):
        return TriState("ABORT", "NO_TECHNICAL_DATA")
    if not flow or not flow.get("ok"):
        return TriState("ABORT", "NO_FLOW_DATA")

    # ── 2. Earnings blackout ─────────────────────────────────────────────
    if fund and fund.get("ok"):
        days_to_er = fund.get("days_to_earnings")
        blackout = cfg.get("fundamental", {}).get("earnings_blackout_days", 2)
        if days_to_er is not None and 0 <= days_to_er <= blackout:
            return TriState("ABORT", f"EARNINGS_BLACKOUT_{days_to_er}D")

    # ── 3. Universe / liquidity floor ────────────────────────────────────
    uni = cfg.get("universe", {})
    price = tech.get("close", 0)
    if price < uni.get("price_min", 1.0):
        return TriState("ABORT", "PRICE_BELOW_FLOOR")
    if price > uni.get("price_max", 500.0):
        return TriState("ABORT", "PRICE_ABOVE_CEILING")

    dollar_vol = flow.get("dollar_volume", 0)
    if dollar_vol < cfg.get("flow_motion", {}).get("dollar_volume_min", 5_000_000):
        return TriState("ABORT", "DOLLAR_VOL_TOO_LOW")

    # ── 4. Core technical floor (HOLD-able) ──────────────────────────────
    tcfg = cfg.get("technical", {})
    if tech.get("rvol", 0) < tcfg.get("rvol_min", 2.0):
        return TriState("HOLD", "RVOL_BELOW_MIN")
    if tech.get("atr_pct", 0) < tcfg.get("atr_pct_min", 0.03):
        return TriState("HOLD", "ATR_BELOW_MIN")

    # ── 5. EMA stack / RSI band ──────────────────────────────────────────
    if tcfg.get("ema_stack_required", True) and not tech.get("ema_stack_ok"):
        return TriState("HOLD", "EMA_STACK_BROKEN")
    rsi = tech.get("rsi", 50)
    rsi_lo, rsi_hi = tcfg.get("rsi_band", [40, 80])
    if not (rsi_lo <= rsi <= rsi_hi):
        return TriState("HOLD", f"RSI_OUT_OF_BAND_{rsi:.1f}")

    # ── 6. Indicator gates ───────────────────────────────────────────────
    icfg = cfg.get("indicators", {})
    if indicators and indicators.get("ok"):
        stack = indicators.get("ma_stack", {}).get("bull_stack_score", 0)
        if icfg.get("require_bull_stack") and stack < icfg.get("min_bull_stack", 0.75):
            return TriState("HOLD", f"BULL_STACK_WEAK_{stack:.2f}")

        heat = indicators.get("stoch_heatmap", {}).get("stoch_bull_heat", 0)
        if heat < icfg.get("min_stoch_bull_heat", 0.0):
            return TriState("HOLD", "STOCH_HEAT_LOW")

        if icfg.get("avoid_overbought"):
            ob = indicators.get("stoch_heatmap", {}).get("overbought_count", 0)
            if ob >= 10:
                return TriState("HOLD", "STOCH_OVERBOUGHT")

        if icfg.get("require_macd_above_zero"):
            if not indicators.get("macd_classic", {}).get("above_zero"):
                return TriState("HOLD", "MACD_BELOW_ZERO")

        if icfg.get("require_above_cloud"):
            if not indicators.get("ichimoku", {}).get("above_cloud"):
                return TriState("HOLD", "BELOW_ICHIMOKU_CLOUD")

        comp = indicators.get("composite_bias", {}).get("bull_score", 0)
        if comp < icfg.get("min_composite_bull", 0.0):
            return TriState("HOLD", f"COMPOSITE_BIAS_WEAK_{comp:.2f}")

    # ── 7. Options flow gate ─────────────────────────────────────────────
    if options and options.get("ok"):
        min_bull = cfg.get("options", {}).get("min_flow_bull_score", 0.0)
        if min_bull > 0 and options.get("flow_bull_score", 0) < min_bull:
            return TriState("HOLD", f"OPTIONS_FLOW_WEAK_{options.get('flow_bull_score',0):.2f}")

    # ── 8. Sector headwind veto ──────────────────────────────────────────
    if sector_ctx and sector_ctx.get("ok"):
        if icfg.get("veto_sector_headwind") and sector_ctx.get("headwind"):
            return TriState("HOLD", "SECTOR_HEADWIND")

    return TriState("GO", "ALL_GATES_PASS")
```

```python
def composite_score(tech, flow, fund, mc, cfg,
                    indicators=None, options=None, sector_ctx=None):
    w = cfg["scoring_weights"]

    t_score = 0.0
    if tech and tech.get("ok"):
        t_score = (min(1.0, tech["rvol"] / 5) * 0.35 +
                   min(1.0, tech["atr_pct"] / 0.10) * 0.25 +
                   (1.0 if tech.get("ema_stack_ok") else 0.0) * 0.20 +
                   (1.0 if tech.get("breakout") else 0.0) * 0.20)

    f_score = 0.0
    if flow and flow.get("ok"):
        f_score = (min(1.0, flow["dollar_volume"] / 50_000_000) * 0.5 +
                   min(1.0, max(0, flow.get("rel_strength_spy", 0)) / 3) * 0.3 +
                   min(1.0, flow.get("news_velocity", 0)) * 0.2)

    fu_score = 0.0
    if fund and fund.get("ok"):
        si = fund.get("short_interest") or 0
        fu_score = min(1.0, si / 0.3) * 0.5 + (0.5 if fund.get("float") else 0.0)

    mc_score = max(0.0, min(1.0, mc["expected_R"] / 1.0)) if mc else 0.0

    ind_score = 0.0
    if indicators and indicators.get("ok"):
        stack = indicators["ma_stack"]["bull_stack_score"]
        heat = indicators["stoch_heatmap"]["stoch_bull_heat"]
        macd_bonus = 0.15 if indicators["macd_classic"]["bull_cross"] else 0.0
        macd_bonus += 0.10 if indicators["macd_classic"]["hist_expanding"] else 0.0
        ich_bonus = indicators["ichimoku"]["bull_strength"] * 0.25
        squeeze_bonus = 0.15 if indicators["bands"]["squeeze"]["just_released"] else 0.0
        div_bonus = 0.10 if indicators["stoch_divergence"]["bull_regular"] else 0.0
        comp = indicators["composite_bias"]["bull_score"]
        ind_score = min(1.0, stack * 0.20 + heat * 0.15 + comp * 0.20 +
                        macd_bonus + ich_bonus + squeeze_bonus + div_bonus)

    opt_score = 0.0
    if options and options.get("ok"):
        opt_score = float(options.get("flow_bull_score", 0.0))

    sect_score = 0.5
    if sector_ctx and sector_ctx.get("ok"):
        if sector_ctx.get("tailwind"):
            sect_score = 1.0
        elif sector_ctx.get("headwind"):
            sect_score = 0.1
        if sector_ctx.get("regime") == "RISK_ON":
            sect_score = min(1.0, sect_score + 0.15)

    w_ind  = float(w.get("indicators", 0.15))
    w_opt  = float(w.get("options", 0.10))
    w_sect = float(w.get("sector", 0.05))
    base = (float(w.get("technical", 0.20)) + float(w.get("flow", 0.15)) +
            float(w.get("fundamental", 0.10)) + float(w.get("monte_carlo", 0.25)))
    overhead = w_ind + w_opt + w_sect

    # ── Guarded scale: never negative, never blow up ─────────────────────
    if base <= 0:
        scale = 0.0
    else:
        scale = max(0.0, (1.0 - overhead)) / base

    raw = (w["technical"] * scale * t_score +
           w["flow"] * scale * f_score +
           w["fundamental"] * scale * fu_score +
           w["monte_carlo"] * scale * mc_score +
           w_ind * ind_score +
           w_opt * opt_score +
           w_sect * sect_score)
    return float(max(0.0, min(1.0, raw)))
```

## 3. `options_flow.py` — FIXED bull score math + labeled RV-proxy

```python
"""
Options chain layer — IV rank (RV-PROXY), unusual options activity, flow scoring.

NOTE: IV rank is currently RV-PROXY (realized vol over 252d). True IV history
requires Tradier historical IV endpoint or persistent IV cache. The output
field `iv_rank_source` documents which method was used.

Bull score components (sum to ≤ 1.0, all bounded [0,1]):
  - 0.25 × pcr_score           (low PCR is bullish)
  - 0.25 × net_premium_score   ([-1,1] mapped to [0,1])
  - 0.20 × call_uoa_share      (call-heavy UOA dominance)
  - 0.15 × iv_sweet_spot       (peaks at IV rank ≈ 0.5)
  - 0.15 × sweep_density       (capped at 5 big sweeps)
"""
from __future__ import annotations
import os, math
from typing import Optional, Dict, List

import numpy as np
import pandas as pd
import requests


# ════════════════════════════════════════════════════════════════════════════
# CHAIN PROVIDERS
# ════════════════════════════════════════════════════════════════════════════

def _tradier_chain(ticker: str) -> Optional[pd.DataFrame]:
    key = os.getenv("TRADIER_API_KEY")
    if not key:
        return None
    base = os.getenv("TRADIER_BASE_URL", "https://sandbox.tradier.com/v1")
    h = {"Authorization": f"Bearer {key}", "Accept": "application/json"}
    try:
        r = requests.get(f"{base}/markets/options/expirations",
                         params={"symbol": ticker}, headers=h, timeout=10)
        exps = r.json().get("expirations", {}).get("date", [])
        if not exps:
            return None
        exps = exps[:4] if isinstance(exps, list) else [exps]
        rows = []
        for exp in exps:
            r = requests.get(f"{base}/markets/options/chains",
                             params={"symbol": ticker, "expiration": exp,
                                     "greeks": "true"},
                             headers=h, timeout=15)
            opts = r.json().get("options", {}).get("option", [])
            if not opts:
                continue
            for o in opts:
                rows.append({
                    "expiration": exp,
                    "strike": o.get("strike"),
                    "type": o.get("option_type"),
                    "bid": o.get("bid"), "ask": o.get("ask"),
                    "last": o.get("last"),
                    "volume": o.get("volume", 0) or 0,
                    "open_interest": o.get("open_interest", 0) or 0,
                    "iv": (o.get("greeks") or {}).get("mid_iv"),
                    "delta": (o.get("greeks") or {}).get("delta"),
                    "gamma": (o.get("greeks") or {}).get("gamma"),
                    "theta": (o.get("greeks") or {}).get("theta"),
                    "vega": (o.get("greeks") or {}).get("vega"),
                })
        return pd.DataFrame(rows) if rows else None
    except Exception:
        return None


def _yf_chain(ticker: str) -> Optional[pd.DataFrame]:
    try:
        import yfinance as yf
        t = yf.Ticker(ticker)
        exps = t.options[:4]
        if not exps:
            return None
        rows = []
        for exp in exps:
            ch = t.option_chain(exp)
            for df, kind in [(ch.calls, "call"), (ch.puts, "put")]:
                for _, r in df.iterrows():
                    rows.append({
                        "expiration": exp, "strike": r["strike"], "type": kind,
                        "bid": r.get("bid", 0), "ask": r.get("ask", 0),
                        "last": r.get("lastPrice", 0),
                        "volume": int(r.get("volume", 0) or 0),
                        "open_interest": int(r.get("openInterest", 0) or 0),
                        "iv": r.get("impliedVolatility"),
                        "delta": None, "gamma": None, "theta": None, "vega": None,
                    })
        return pd.DataFrame(rows) if rows else None
    except Exception:
        return None


def get_chain(ticker: str) -> Optional[pd.DataFrame]:
    for fn in (_tradier_chain, _yf_chain):
        df = fn(ticker)
        if df is not None and not df.empty:
            return df
    return None


# ════════════════════════════════════════════════════════════════════════════
# IV RANK / PERCENTILE — RV-PROXY (clearly labeled)
# ════════════════════════════════════════════════════════════════════════════

def iv_rank_percentile(ticker: str, current_iv: float) -> Dict:
    """RV-PROXY: compares current IV against 252d realized-vol distribution.
    Not a true IV rank. Field `iv_rank_source` = "RV_PROXY"."""
    try:
        import yfinance as yf
        hist = yf.Ticker(ticker).history(period="1y")
        if hist.empty or current_iv is None:
            return {"iv_rank": None, "iv_percentile": None,
                    "iv_current": current_iv, "iv_rank_source": "UNAVAILABLE"}
        rv = hist["Close"].pct_change().rolling(20).std() * math.sqrt(252)
        rv = rv.dropna()
        if rv.empty:
            return {"iv_rank": None, "iv_percentile": None,
                    "iv_current": current_iv, "iv_rank_source": "UNAVAILABLE"}
        lo, hi = float(rv.min()), float(rv.max())
        rank = (current_iv - lo) / (hi - lo + 1e-12)
        pct = float((rv < current_iv).mean())
        return {
            "iv_rank": float(np.clip(rank, 0, 1)),
            "iv_percentile": pct,
            "iv_current": float(current_iv),
            "rv_20d": float(rv.iloc[-1]),
            "iv_rank_source": "RV_PROXY",
        }
    except Exception:
        return {"iv_rank": None, "iv_percentile": None,
                "iv_current": current_iv, "iv_rank_source": "ERROR"}


# ════════════════════════════════════════════════════════════════════════════
# FLOW METRICS
# ════════════════════════════════════════════════════════════════════════════

def _atm_iv(chain: pd.DataFrame, spot: float) -> Optional[float]:
    front = chain[chain["expiration"] == chain["expiration"].min()]
    if front.empty:
        return None
    front = front.assign(dist=(front["strike"] - spot).abs())
    atm = front.nsmallest(4, "dist")
    iv = atm["iv"].dropna()
    return float(iv.mean()) if not iv.empty else None


def detect_uoa(chain: pd.DataFrame, vol_to_oi_min: float = 3.0,
               min_volume: int = 500) -> pd.DataFrame:
    df = chain.copy()
    df["vol_oi"] = df["volume"] / (df["open_interest"] + 1)
    uoa = df[(df["vol_oi"] >= vol_to_oi_min) & (df["volume"] >= min_volume)].copy()
    uoa["notional"] = uoa["volume"] * (uoa["last"].fillna(0)) * 100
    return uoa.sort_values("notional", ascending=False)


def compute_options_flow(ticker: str, spot: float,
                         cfg: Optional[Dict] = None) -> Dict:
    cfg = cfg or {}
    chain = get_chain(ticker)
    if chain is None or chain.empty:
        return {"ok": False, "reason": "NO_CHAIN_DATA"}

    chain = chain.dropna(subset=["strike", "type"])
    chain["volume"] = chain["volume"].fillna(0).astype(int)
    chain["open_interest"] = chain["open_interest"].fillna(0).astype(int)
    chain["last"] = chain["last"].fillna(0).astype(float)

    calls = chain[chain["type"] == "call"]
    puts  = chain[chain["type"] == "put"]

    call_vol, put_vol = int(calls["volume"].sum()), int(puts["volume"].sum())
    call_oi,  put_oi  = int(calls["open_interest"].sum()), int(puts["open_interest"].sum())

    pcr_vol = put_vol / max(call_vol, 1)
    pcr_oi  = put_oi  / max(call_oi, 1)

    call_prem = float((calls["volume"] * calls["last"] * 100).sum())
    put_prem  = float((puts["volume"]  * puts["last"]  * 100).sum())
    net_prem  = call_prem - put_prem
    total_prem = call_prem + put_prem

    atm_iv = _atm_iv(chain, spot)
    iv_meta = iv_rank_percentile(ticker, atm_iv) if atm_iv else {
        "iv_rank": None, "iv_percentile": None,
        "iv_current": None, "iv_rank_source": "NO_ATM_IV"}

    uoa = detect_uoa(chain,
                     vol_to_oi_min=cfg.get("uoa_vol_oi_min", 3.0),
                     min_volume=cfg.get("uoa_min_volume", 500))
    uoa_call = uoa[uoa["type"] == "call"]
    uoa_put  = uoa[uoa["type"] == "put"]

    big_sweeps = uoa[uoa["notional"] >= cfg.get("sweep_min_notional", 50_000)]
    sweeps_list = [
        {"type": r["type"], "strike": float(r["strike"]),
         "exp": r["expiration"], "vol": int(r["volume"]),
         "oi": int(r["open_interest"]),
         "notional": round(float(r["notional"]), 0)}
        for _, r in big_sweeps.head(10).iterrows()
    ]

    # ── FIXED Composite flow score ──────────────────────────────────────
    # Each component bounded [0,1]; weights sum to 1.0
    pcr_score = 1.0 if pcr_vol < 0.7 else (0.4 if pcr_vol < 1.0 else 0.0)

    if total_prem > 0:
        net_ratio = np.clip(net_prem / total_prem, -1.0, 1.0)
        net_score = (net_ratio + 1.0) / 2.0     # [-1,1] → [0,1]
    else:
        net_score = 0.5

    if len(uoa) > 0 and uoa["notional"].sum() > 0:
        call_uoa_share = float(uoa_call["notional"].sum() / uoa["notional"].sum())
    else:
        call_uoa_share = 0.5

    if iv_meta.get("iv_rank") is not None:
        # Sweet spot at IV rank 0.5 (moderate IV); penalize extremes
        iv_sweet = float(1.0 - abs(iv_meta["iv_rank"] - 0.5) * 2.0)
        iv_sweet = max(0.0, iv_sweet)
    else:
        iv_sweet = 0.5

    sweep_density = float(min(1.0, len(big_sweeps) / 5.0))

    bull = (0.25 * pcr_score +
            0.25 * net_score +
            0.20 * call_uoa_share +
            0.15 * iv_sweet +
            0.15 * sweep_density)
    bull_score = float(np.clip(bull, 0.0, 1.0))

    return {
        "ok": True,
        "pcr_volume": round(pcr_vol, 3),
        "pcr_oi": round(pcr_oi, 3),
        "call_volume": call_vol,
        "put_volume": put_vol,
        "call_premium": round(call_prem, 0),
        "put_premium": round(put_prem, 0),
        "net_premium": round(net_prem, 0),
        "total_premium": round(total_prem, 0),
        "atm_iv": atm_iv,
        "iv_rank": iv_meta.get("iv_rank"),
        "iv_percentile": iv_meta.get("iv_percentile"),
        "iv_rank_source": iv_meta.get("iv_rank_source"),
        "uoa_count": int(len(uoa)),
        "uoa_call_count": int(len(uoa_call)),
        "uoa_put_count": int(len(uoa_put)),
        "big_sweeps": sweeps_list,
        "flow_bull_score": round(bull_score, 3),
        "flow_bear_score": round(1.0 - bull_score, 3),
        "score_components": {
            "pcr_score": round(pcr_score, 3),
            "net_score": round(net_score, 3),
            "call_uoa_share": round(call_uoa_share, 3),
            "iv_sweet": round(iv_sweet, 3),
            "sweep_density": round(sweep_density, 3),
        },
    }
```

## 4. `broker_alpaca.py` — FIXED (min_score gate + daily loss circuit + quote refresh + version-safe orders)

```python
"""
Alpaca paper-trading execution layer (hardened MVP).

SAFETY GATES (all enforced):
  - PAPER default (ALPACA_PAPER=true)
  - max_position_usd, max_concurrent (config + env)
  - min_score_to_fire (config-driven)
  - Daily loss circuit breaker (max_daily_loss_pct)
  - Idempotency (open positions + open orders)
  - Live quote refresh before sizing (slippage protection)
"""
from __future__ import annotations
import os
from dataclasses import dataclass
from typing import Optional, Dict, List

import structlog

log = structlog.get_logger("broker")


@dataclass
class ExecutionResult:
    ok: bool
    ticker: str
    order_id: Optional[str] = None
    qty: int = 0
    entry: float = 0.0
    target: float = 0.0
    stop: float = 0.0
    reason: str = ""


class AlpacaBroker:
    def __init__(self, paper: bool = True):
        from alpaca.trading.client import TradingClient
        self.paper = paper or os.getenv("ALPACA_PAPER", "true").lower() == "true"
        key = os.getenv("ALPACA_API_KEY")
        sec = os.getenv("ALPACA_SECRET_KEY")
        if not (key and sec):
            raise RuntimeError("ALPACA_API_KEY / ALPACA_SECRET_KEY missing")
        self.client = TradingClient(key, sec, paper=self.paper)
        self._data_client = None

    def _data(self):
        if self._data_client is None:
            from alpaca.data.historical import StockHistoricalDataClient
            self._data_client = StockHistoricalDataClient(
                os.getenv("ALPACA_API_KEY"), os.getenv("ALPACA_SECRET_KEY"))
        return self._data_client

    # ── Account ─────────────────────────────────────────────────────────
    def account(self) -> Dict:
        a = self.client.get_account()
        return {
            "equity": float(a.equity),
            "last_equity": float(getattr(a, "last_equity", a.equity)),
            "buying_power": float(a.buying_power),
            "cash": float(a.cash),
            "portfolio_value": float(a.portfolio_value),
            "paper": self.paper,
        }

    def open_positions(self) -> List[str]:
        try:
            return [p.symbol for p in self.client.get_all_positions()]
        except Exception as e:
            log.warning("positions_fetch_failed", error=str(e))
            return []

    def open_orders(self) -> List[str]:
        """Version-safe: tries new alpaca-py first, falls back."""
        try:
            from alpaca.trading.requests import GetOrdersRequest
            from alpaca.trading.enums import QueryOrderStatus
            try:
                req = GetOrdersRequest(status=QueryOrderStatus.OPEN)
                orders = self.client.get_orders(filter=req)
            except TypeError:
                orders = self.client.get_orders(req)
            return [o.symbol for o in orders]
        except Exception as e:
            log.warning("orders_fetch_failed", error=str(e))
            return []

    # ── Live quote (slippage protection) ────────────────────────────────
    def latest_quote(self, ticker: str) -> Optional[float]:
        try:
            from alpaca.data.requests import StockLatestQuoteRequest
            req = StockLatestQuoteRequest(symbol_or_symbols=ticker)
            q = self._data().get_stock_latest_quote(req)
            row = q.get(ticker)
            if row is None:
                return None
            ask = float(getattr(row, "ask_price", 0) or 0)
            bid = float(getattr(row, "bid_price", 0) or 0)
            if ask > 0 and bid > 0:
                return (ask + bid) / 2.0
            return ask or bid or None
        except Exception as e:
            log.warning("quote_fetch_failed", ticker=ticker, error=str(e))
            return None

    # ── Order placement ─────────────────────────────────────────────────
    def submit_bracket(self, ticker: str, qty: int, entry: float,
                       target: float, stop: float,
                       side: str = "buy") -> ExecutionResult:
        from alpaca.trading.requests import (
            MarketOrderRequest, TakeProfitRequest, StopLossRequest)
        from alpaca.trading.enums import OrderSide, TimeInForce, OrderClass
        try:
            # Sanity: ensure stop < entry < target for long
            if side == "buy" and not (stop < entry < target):
                return ExecutionResult(ok=False, ticker=ticker,
                                       reason=f"INVALID_LEVELS s={stop} e={entry} t={target}")
            req = MarketOrderRequest(
                symbol=ticker, qty=qty,
                side=OrderSide.BUY if side == "buy" else OrderSide.SELL,
                time_in_force=TimeInForce.DAY,
                order_class=OrderClass.BRACKET,
                take_profit=TakeProfitRequest(limit_price=round(target, 2)),
                stop_loss=StopLossRequest(stop_price=round(stop, 2)),
            )
            o = self.client.submit_order(req)
            log.info("order_submitted", ticker=ticker, qty=qty,
                     entry=entry, target=target, stop=stop, id=str(o.id))
            return ExecutionResult(
                ok=True, ticker=ticker, order_id=str(o.id),
                qty=qty, entry=entry, target=target, stop=stop,
                reason="SUBMITTED",
            )
        except Exception as e:
            log.error("order_failed", ticker=ticker, error=str(e))
            return ExecutionResult(ok=False, ticker=ticker, reason=f"ERROR: {e}")


def auto_execute(payload: Dict, cfg: Dict,
                 dry_run: bool = False) -> List[ExecutionResult]:
    """Execute GO candidates with full safety gates."""
    results: List[ExecutionResult] = []
    candidates = payload.get("candidates", [])
    if not candidates:
        return results

    exec_cfg = cfg.get("execution", {})
    risk = cfg.get("risk", {})
    min_score = float(exec_cfg.get("min_score_to_fire", 0.50))
    max_pos_usd = float(os.getenv("MAX_POSITION_USD",
                                  risk.get("max_position_usd", 1000)))
    max_conc = int(os.getenv("MAX_CONCURRENT_POSITIONS",
                             risk.get("max_concurrent_positions", 5)))
    max_daily_loss = float(risk.get("max_daily_loss_pct", 0.02))
    slip_buffer = float(exec_cfg.get("slippage_buffer_pct", 0.005))

    # ── Score gate (apply BEFORE broker init) ────────────────────────────
    qualified = [c for c in candidates if c.get("score", 0) >= min_score]
    if not qualified:
        return [ExecutionResult(ok=False, ticker="*",
                                reason=f"NO_CANDIDATES_ABOVE_SCORE_{min_score}")]

    # ── Broker init ──────────────────────────────────────────────────────
    broker = None
    if not dry_run:
        try:
            broker = AlpacaBroker(paper=True)
            acct = broker.account()
        except Exception as e:
            log.error("broker_init_failed", error=str(e))
            return [ExecutionResult(ok=False, ticker="*", reason=f"INIT: {e}")]
    else:
        acct = {"equity": 100_000.0, "last_equity": 100_000.0,
                "buying_power": 100_000.0}

    # ── Daily loss circuit breaker ───────────────────────────────────────
    eq, last_eq = acct["equity"], acct.get("last_equity", acct["equity"])
    if last_eq > 0:
        day_pnl_pct = (eq - last_eq) / last_eq
        if day_pnl_pct < -max_daily_loss:
            return [ExecutionResult(
                ok=False, ticker="*",
                reason=f"DAILY_LOSS_LIMIT_HIT {day_pnl_pct:.2%} < -{max_daily_loss:.2%}")]

    # ── Idempotency ──────────────────────────────────────────────────────
    held = set()
    if broker:
        held = set(broker.open_positions()) | set(broker.open_orders())
    free_slots = max(0, max_conc - len(held))
    if free_slots == 0:
        return [ExecutionResult(ok=False, ticker="*", reason="MAX_CONCURRENT_REACHED")]

    buying_power = acct["buying_power"]

    for cand in qualified[:free_slots]:
        tk = cand["ticker"]
        if tk in held:
            results.append(ExecutionResult(ok=False, ticker=tk, reason="ALREADY_HELD"))
            continue

        mc = cand.get("monte_carlo")
        if not mc:
            results.append(ExecutionResult(ok=False, ticker=tk, reason="NO_MC_LEVELS"))
            continue

        # ── Refresh quote for slippage protection ────────────────────────
        stale_close = float(cand["technical"]["close"])
        live = broker.latest_quote(tk) if broker else None
        entry = live if (live and live > 0) else stale_close
        # Apply small upward slippage buffer for market-buy limit cap
        entry_with_slip = entry * (1.0 + slip_buffer)

        target = float(mc.get("target_price", entry * 1.02))
        stop   = float(mc.get("stop_price",   entry * 0.99))

        # Re-anchor target/stop if quote drifted >2% from close
        if live and abs(live - stale_close) / stale_close > 0.02:
            R = stale_close - float(mc.get("stop_price", stale_close * 0.99))
            if R > 0:
                stop = live - R
                target = live + R * cfg.get("monte_carlo", {}).get("target_R", 2.0)

        usd_per = min(max_pos_usd, buying_power * 0.95)
        qty = int(usd_per // entry_with_slip)
        if qty <= 0:
            results.append(ExecutionResult(ok=False, ticker=tk, reason="QTY_ZERO"))
            continue

        if dry_run:
            results.append(ExecutionResult(
                ok=True, ticker=tk, qty=qty, entry=entry,
                target=target, stop=stop, reason="DRY_RUN"))
            buying_power -= qty * entry
            continue

        res = broker.submit_bracket(tk, qty, entry, target, stop)
        results.append(res)
        if res.ok:
            buying_power -= qty * entry

    return results
```

```python
def scan(tickers, cfg, benchmark="SPY", return_dataframes=False,
         compute_options: bool = True, compute_sectors: bool = True):
    router = DataRouter()
    bench = _safe_bench(router, benchmark)
    rotation = compute_sector_rotation(router) if compute_sectors else {"ok": False}

    results = []
    for tk in tickers:
        try:
            d_res = router.get_daily(tk, 250)
            if not (d_res and d_res.ok and d_res.data is not None and not d_res.data.empty):
                results.append({"ticker": tk, "verdict": "ABORT",
                                "reason": "NO_DAILY_DATA", "score": 0.0})
                continue

                   {"ok": False, "reason": "NO_BENCHMARK"}

            options = (compute_options_flow(tk, tech.get("close", 0), cfg.get("options", {}))
                       if compute_options else {"ok": False})
            sect_ctx = ticker_sector_context(tk, rotation) if rotation.get("ok") else {"ok": False}

            q = qualify(tech, fund, flow, cfg, indicators, options, sect_ctx)

            mc, mc_v = None, "SKIP"
            if q.state == "GO":
                ic = i_res.data["close"] if (i_res and i_res.ok) else None
                mc = run_mc(tech["close"], tech["atr_pct"], ic, cfg,
                            has_catalyst=len(news) >= 3)
                mc_v = mc_verdict(mc, cfg)

            final, reason = q.state, q.reason
            if q.state == "GO":
                if mc_v == "ABORT": final, reason = "ABORT", "MC_TAIL_RISK"
                elif mc_v == "HOLD": final, reason = "HOLD", "MC_EDGE_WEAK"

            score = composite_score(tech, flow, fund, mc, cfg,
                                    indicators, options, sect_ctx) \
                    if final == "GO" else 0.0

            ind_out = {k: v for k, v in indicators.items() if k != "_raw"} \
                      if indicators else {}

            entry = {
                "ticker": tk, "verdict": final, "reason": reason,
                "score": round(score, 4),
                "technical": tech, "fundamental": fund, "flow": flow,
                "indicators": ind_out, "monte_carlo": mc,
                "options_flow": options, "sector_context": sect_ctx,
                "provider_daily": d_res.provider,
                "provider_intraday": i_res.provider if (i_res and i_res.ok) else None,
            }
            if return_dataframes:
                entry["_daily_df"] = d_res.data
                entry["_intraday_df"] = i_res.data if (i_res and i_res.ok) else None

    }
```

```python
"""Sector rotation heatmap + lazy yfinance sector resolution with on-disk cache."""
from __future__ import annotations
import os, json
from typing import Dict, Optional
import pandas as pd
import numpy as np

SECTOR_ETFS = {
    "XLK": "Technology", "XLY": "Consumer Discretionary",
    "XLC": "Communication Services", "XLF": "Financials",
    "XLV": "Healthcare", "XLI": "Industrials",
    "XLP": "Consumer Staples", "XLU": "Utilities",
    "XLE": "Energy", "XLB": "Materials", "XLRE": "Real Estate",
}

YF_SECTOR_TO_ETF = {
    "Technology": "XLK", "Consumer Cyclical": "XLY", "Consumer Defensive": "XLP",
    "Communication Services": "XLC", "Financial Services": "XLF",
    "Healthcare": "XLV", "Industrials": "XLI", "Utilities": "XLU",
    "Energy": "XLE", "Basic Materials": "XLB", "Real Estate": "XLRE",
}

STATIC_MAP = {
    "AAPL": "XLK", "MSFT": "XLK", "NVDA": "XLK", "AMD": "XLK",
    "GOOGL": "XLC", "META": "XLC", "NFLX": "XLC", "DIS": "XLC",
    "AMZN": "XLY", "TSLA": "XLY", "HD": "XLY", "NKE": "XLY",
    "JPM": "XLF", "BAC": "XLF", "GS": "XLF", "COIN": "XLF",
    "UNH": "XLV", "JNJ": "XLV", "PFE": "XLV", "LLY": "XLV",
    "XOM": "XLE", "CVX": "XLE", "OXY": "XLE",
    "WMT": "XLP", "PG": "XLP", "KO": "XLP",
    "PLTR": "XLK", "SOFI": "XLF", "RIOT": "XLK", "MARA": "XLK",
}

CACHE_PATH = os.path.join("logs", "sector_cache.json")


def _load_cache() -> Dict[str, str]:
    try:
        with open(CACHE_PATH) as f:
            return json.load(f)
    except Exception:
        return {}


def _save_cache(c: Dict[str, str]):
    try:
        os.makedirs(os.path.dirname(CACHE_PATH), exist_ok=True)
        with open(CACHE_PATH, "w") as f:
            json.dump(c, f)
    except Exception:
        pass


def _resolve_sector(ticker: str) -> Optional[str]:
    tk = ticker.upper()
    if tk in STATIC_MAP:
        return STATIC_MAP[tk]
    cache = _load_cache()
    if tk in cache:
        return cache[tk] or None
    try:
        import yfinance as yf
        info = yf.Ticker(tk).info or {}
        yf_sector = info.get("sector")
        etf = YF_SECTOR_TO_ETF.get(yf_sector)
        cache[tk] = etf or ""
        _save_cache(cache)
        return etf
    except Exception:
        cache[tk] = ""
        _save_cache(cache)
        return None


def fetch_sector_data(router, days: int = 60) -> pd.DataFrame:
    syms = list(SECTOR_ETFS.keys()) + ["SPY"]
    out = {}
    for s in syms:
        try:
            res = router.get_daily(s, days)
            if res and res.ok and not res.data.empty:
                out[s] = res.data["close"]
        except Exception:
            pass
    return pd.DataFrame(out).dropna(how="all")


def compute_sector_rotation(router) -> Dict:
    df = fetch_sector_data(router)
    if df.empty or "SPY" not in df.columns:
        return {"ok": False, "reason": "NO_SECTOR_DATA"}

    spy_ret = {
        "1d": float(df["SPY"].pct_change(1).iloc[-1]),
        "5d": float(df["SPY"].pct_change(5).iloc[-1]),
        "20d": float(df["SPY"].pct_change(20).iloc[-1]),
    }

    sectors = []
    for etf, name in SECTOR_ETFS.items():
        if etf not in df.columns:
            continue
        s = df[etf]
        r1, r5, r20 = (float(s.pct_change(n).iloc[-1]) for n in (1, 5, 20))
        rs1 = r1 - spy_ret["1d"]
        rs5 = r5 - spy_ret["5d"]
        rs20 = r20 - spy_ret["20d"]
        sectors.append({
            "etf": etf, "name": name,
            "ret_1d": round(r1 * 100, 2), "ret_5d": round(r5 * 100, 2),
            "ret_20d": round(r20 * 100, 2),
            "rs_1d": round(rs1 * 100, 2), "rs_5d": round(rs5 * 100, 2),
            "rs_20d": round(rs20 * 100, 2),
            "leader": rs5 > 0 and rs20 > 0,
            "laggard": rs5 < 0 and rs20 < 0,
        })
    sectors.sort(key=lambda x: x["rs_5d"], reverse=True)

    cyclical = ["XLK", "XLY", "XLC", "XLF", "XLI"]
    defensive = ["XLP", "XLU", "XLV", "XLRE"]
    cyc_rs = float(np.mean([s["rs_5d"] for s in sectors if s["etf"] in cyclical]) or 0)
    def_rs = float(np.mean([s["rs_5d"] for s in sectors if s["etf"] in defensive]) or 0)

    if cyc_rs > def_rs + 0.3:   regime = "RISK_ON"
    elif def_rs > cyc_rs + 0.3: regime = "RISK_OFF"
    else:                       regime = "NEUTRAL"

    return {
        "ok": True, "regime": regime,
        "cyclical_rs": round(cyc_rs, 2), "defensive_rs": round(def_rs, 2),
        "spy_returns": spy_ret, "sectors": sectors,
        "leaders": [s for s in sectors if s["leader"]][:3],
        "laggards": [s for s in sectors if s["laggard"]][-3:],
    }


def ticker_sector_context(ticker: str, rotation: Dict) -> Dict:
    if not rotation.get("ok"):
        return {"ok": False}
    etf = _resolve_sector(ticker)
    if not etf:
        return {"ok": False, "reason": "UNMAPPED_SECTOR"}
    sec = next((s for s in rotation["sectors"] if s["etf"] == etf), None)
    if not sec:
        return {"ok": False, "reason": "SECTOR_NOT_IN_ROTATION"}
    return {
        "ok": True,
        "sector_etf": etf, "sector_name": sec["name"],
        "rs_5d": sec["rs_5d"], "rs_20d": sec["rs_20d"],
        "tailwind": sec["leader"], "headwind": sec["laggard"],
        "regime": rotation["regime"],
    }
```
